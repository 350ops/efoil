        const mobileMenuBtn = document.getElementById('mobile-menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');

        // --- INITIALIZATION ---
        document.addEventListener('DOMContentLoaded', () => {
            initCharts();
            updateDashboard(0); // Start at Jan
            renderEvents('all');
            renderLocations();
            
            // Event Listeners
            monthSlider.addEventListener('input', (e) => {
                updateDashboard(parseInt(e.target.value));
            });

            document.querySelectorAll('.filter-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    // Update Active State
                    document.querySelectorAll('.filter-btn').forEach(b => {
                        b.classList.remove('bg-ocean-800', 'text-white');
                        b.classList.add('bg-sand-200', 'text-ocean-900');
                    });
                    e.target.classList.remove('bg-sand-200', 'text-ocean-900');
                    e.target.classList.add('bg-ocean-800', 'text-white');
                    
                    renderEvents(e.target.dataset.filter);
                });
            });

            mobileMenuBtn.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });
        });

        // --- CORE LOGIC ---

        function updateDashboard(monthIndex) {
            const data = appData.monthlyData[monthIndex];

            // Text Updates
            monthDisplay.textContent = data.name.toUpperCase();
            conditionDisplay.textContent = data.desc;
            windDisplay.textContent = data.wind;
            
            // Bar Width Update
            const maxWind = 20; // Scale reference
            const width = (data.wind / maxWind) * 100;
            windBar.style.width = `${width}%`;

            // Tags Update
            recommendedSports.innerHTML = data.sports.map(s => 
                `<span class="px-2 py-1 bg-lagoon-300/20 text-lagoon-500 text-xs font-bold rounded-md border border-lagoon-300/30">${s}</span>`
            ).join('');

            // Chart Updates
            updateRadarChart(data);
            updateBarChart(data);
        }

        function initCharts() {
            // 1. Radar Chart: Conditions
            const ctxRadar = document.getElementById('conditionsRadar').getContext('2d');
            radarChart = new Chart(ctxRadar, {
                type: 'radar',
                data: {
                    labels: ['Wind Strength', 'Swell Height', 'Sun Hours', 'Rain (Inverse)'],
                    datasets: [{
                        label: 'Current Month',
                        data: [0, 0, 0, 0], // Init placeholder
                        backgroundColor: 'rgba(6, 182, 212, 0.4)', // lagoon-500
                        borderColor: '#06b6d4',
                        pointBackgroundColor: '#1e3a8a',
                        borderWidth: 2
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        r: {
                            angleLines: { color: '#ebe5d5' },
                            grid: { color: '#ebe5d5' },
                            pointLabels: { font: { size: 12, family: 'Inter' }, color: '#1e3a8a' },
                            suggestedMin: 0,
                            suggestedMax: 10
                        }
                    },
                    plugins: { legend: { display: false } }
                }
            });

            // 2. Bar Chart: Suitability
            const ctxBar = document.getElementById('suitabilityBar').getContext('2d');
            barChart = new Chart(ctxBar, {
                type: 'bar',
                data: {
                    labels: ['Kite/Wing', 'Surf', 'E-Foil/Chill'],
                    datasets: [{
                        label: 'Suitability Score (0-10)',
                        data: [0, 0, 0],
                        backgroundColor: ['#f43f5e', '#1e3a8a', '#06b6d4'],
                        borderRadius: 5
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        y: { beginAtZero: true, max: 10, grid: { display: false } },
                        x: { grid: { display: false } }
                    },
                    plugins: { legend: { display: false } }
                }
            });

            // 3. Line Chart: Annual Trends (Static)
            const ctxLine = document.getElementById('trendLineChart').getContext('2d');
            new Chart(ctxLine, {
                type: 'line',
                data: {
                    labels: appData.months,
                    datasets: [
                        {
                            label: 'Avg Wind (Knots)',
                            data: appData.monthlyData.map(d => d.wind),
                            borderColor: '#06b6d4',
                            backgroundColor: 'rgba(6, 182, 212, 0.1)',
                            fill: true,
                            tension: 0.4
                        },
                        {
                            label: 'Avg Swell (ft)',
                            data: appData.monthlyData.map(d => d.swell),
                            borderColor: '#1e3a8a',
                            borderDash: [5, 5],
                            tension: 0.4
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    interaction: { mode: 'index', intersect: false },
                    scales: {
                        y: { grid: { color: '#f7f4ea' } },
                        x: { grid: { display: false } }
                    },
                    plugins: {
                        tooltip: {
                            backgroundColor: '#0f172a',
                            titleFont: { family: 'Inter' },
                            bodyFont: { family: 'Inter' }
                        }
                    }
                }
            });
        }

        function updateRadarChart(data) {
            // Normalizing data to 0-10 scale for the chart
            // Wind: Max ~20, Swell: Max ~10, Sun: Max ~10, Rain: Max ~10
            const rainScore = 10 - data.rain; // Inverse because less rain is "better" for general tourism, though maybe not for wind
            const windScore = (data.wind / 20) * 10;
            
            radarChart.data.datasets[0].data = [windScore, data.swell, data.sun, rainScore];
            radarChart.update();
        }

        function updateBarChart(data) {
            // Heuristic scoring based on month data
            let kiteScore = (data.wind > 12) ? (data.wind / 2) : 2; 
            let surfScore = data.swell;
            let chillScore = (data.wind < 12) ? 9 : (10 - (data.wind/3));

            barChart.data.datasets[0].data = [kiteScore, surfScore, chillScore];
            barChart.update();
        }

        function renderEvents(filter) {
            eventsGrid.innerHTML = '';
            
            const filteredEvents = filter === 'all' 
                ? appData.events 
                : appData.events.filter(e => e.type === filter || (filter === 'wind' && (e.type === 'wing' || e.type === 'kite')));

            if (filteredEvents.length === 0) {
                eventsGrid.innerHTML = '<p class="text-ocean-900/50 col-span-2 text-center py-8">No specific events found for this category.</p>';
                return;
            }

            filteredEvents.forEach(event => {
                // Determine icon
                let icon = '🌊';
                if(event.type === 'surf') icon = '🏄‍♂️';
                if(event.type === 'wind' || event.type === 'kite') icon = '🪁';
                if(event.type === 'bodyboard') icon = '🏊';

                const card = document.createElement('div');
                card.className = 'bg-white p-6 rounded-xl border border-sand-200 shadow-sm hover:shadow-md transition flex flex-col md:flex-row gap-4 items-start';
                card.innerHTML = `
                    <div class="bg-lagoon-100 text-3xl p-4 rounded-full flex-shrink-0">${icon}</div>
                    <div>
                        <div class="flex flex-wrap items-center gap-2 mb-1">
                            <span class="text-xs font-bold uppercase text-lagoon-500 tracking-wider">${event.date}</span>
                            <span class="text-xs px-2 py-0.5 bg-sand-200 text-ocean-800 rounded-full">${event.status}</span>
                        </div>
                        <h3 class="text-xl font-bold text-ocean-900 mb-1">${event.title}</h3>
                        <p class="text-sm font-semibold text-ocean-800/60 mb-2">📍 ${event.location}</p>
                        <p class="text-sm text-ocean-900/70 leading-relaxed">${event.desc}</p>
                    </div>
                `;
                eventsGrid.appendChild(card);
            });
        }

        function renderLocations(highlightTag = null) {
            locationGrid.innerHTML = '';
            
            appData.locations.forEach(loc => {
                const isHighlighted = highlightTag ? loc.tags.includes(highlightTag) : true;
                const opacityClass = isHighlighted ? 'opacity-100' : 'opacity-40 grayscale';
                const shadowClass = isHighlighted ? 'shadow-lg ring-2 ring-lagoon-300' : 'shadow-sm';

                const card = document.createElement('div');
                card.className = `bg-white p-6 rounded-xl border border-sand-200 transition duration-300 ${opacityClass} ${shadowClass}`;
                
                const tagsHtml = loc.tags.map(t => `<span class="text-xs font-bold text-ocean-800 bg-sand-200 px-2 py-1 rounded-md uppercase">${t}</span>`).join(' ');

                card.innerHTML = `
                    <div class="flex justify-between items-start mb-4">
                        <h3 class="text-lg font-bold text-ocean-900">${loc.name}</h3>
                        <span class="text-xs text-ocean-500 font-mono">${loc.atoll}</span>
                    </div>
                    <p class="text-sm text-ocean-900/70 mb-4">${loc.desc}</p>
                    <div class="flex gap-2">${tagsHtml}</div>
                `;
                locationGrid.appendChild(card);
            });
        }

        // Exposed function for clicking sport cards
        function highlightLocations(tag) {
            renderLocations(tag);
            // Scroll to locations
            document.getElementById('locations').scrollIntoView({behavior: 'smooth'});
        }
